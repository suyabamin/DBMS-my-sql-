1. Show employee names and salary.

SELECT first_name,last_name,salary
FROM employees;

2. Show employees earning more than 5000.

SELECT first_name,last_name,salary
FROM employees
WHERE salary>5000;

3. Show all job titles.

SELECT job_id,job_title
FROM jobs;

4. Show departments sorted alphabetically.

SELECT department_name
FROM departments
ORDER BY department_name;

5. Show locations in the city starting with 'S'.

SELECT location_id,city
FROM locations
WHERE city LIKE 's%';

6. Count the number of employees per job_id.

SELECT job_id, COUNT(*)
FROM employees
GROUP BY job_id;

7. Find departments having more than 3 employees.

SELECT department_id,COUNT(*)
FROM employees
GROUP BY department_id
HAVING COUNT(*)>3;

8. Show average salary for each department.

SELECT department_id, AVG(salary)
FROM employees
GROUP BY department_id

9. List job_ids where max_salary > 15000.
SELECT job_id, max_salary
FROM jobs
WHERE max_salary > 15000;

10. Count employees hired each year.

SELECT EXTRACT(YEAR FROM hire_date), COUNT(*) 
FROM employees
GROUP BY EXTRACT(YEAR FROM hire_date)
ORDER BY hire_date;

-- 11. Show the number of employees for each salary level (group by salary).

SELECT salary,COUNT(*)
FROM employees
GROUP BY salary
ORDER BY salary DESC;

-- 12. Show job_ids where avg salary (in employees table) is between 5000 and 10000.

SELECT job_id ,AVG(salary)
FROM employees
GROUP BY job_id
HAVING AVG(salary)BETWEEN 5000 AND 10000;

-- 13. Count employees per commission_pct.

SELECT job_id,commission_pct,COUNT(*)
FROM employees
GROUP BY commission_pct;

-- 14. Show years in which more than 10 employees were hired.
SELECT EXTRACT(YEAR FROM hire_date) as hire_year,COUNT(*) as hires
FROM employees
GROUP BY EXTRACT(YEAR FROM hire_date)
HAVING COUNT(*)>10;


-- 15. Find min, max, and avg salary per department.
SELECT department_id, MAX(salary)as max_salary,MIN(salary)as min_salary,AVG(salary)as avg_salary
FROM employees
GROUP BY department_id;
